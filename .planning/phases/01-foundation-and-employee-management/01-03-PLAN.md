    1. Create src/lib/actions/employees.ts ("use server"):
       - Import: auth from @/lib/auth, db from @/lib/db, schema tables, zod, revalidatePath, eq from drizzle-orm, bcrypt from bcryptjs

       - createEmployeeSchema: z.object({
           name: z.string().min(1, "Name is required"),
           email: z.string().email("Valid email required"),
           role: z.enum(["manager", "supervisor", "employee"]),
           hourlyRate: z.coerce.number().positive("Rate must be positive"),
           maxHoursPerWeek: z.coerce.number().int().min(1).max(60),
           phone: z.string().optional(),
           jobRoles: z.array(z.enum(["cashier", "stock", "manager", "visual_merch"])).min(1, "At least one job role required"),
         })

       - updateEmployeeSchema: same as create but with id: z.number()

       - createEmployee(formData: FormData):
         * Check auth() session, require role === "manager"
         * Parse and validate with createEmployeeSchema
         * Hash a default password "changeme123" with bcryptjs
         * Insert into users table (storeId = 1 for single store)
         * Insert job roles into employeeRoles junction table
         * Insert default availability (all days available)
         * revalidatePath("/employees")
         * Return { success: true, employeeId }

       - updateEmployee(formData: FormData):
         * Check auth() session, require role === "manager"
         * Parse with updateEmployeeSchema
         * Update users table by id
         * Delete existing employeeRoles for user, insert new ones
         * revalidatePath("/employees")
         * Return { success: true }

       - updateAvailability(userId: number, dayOfWeek: number, isAvailable: boolean):
         * Check auth() session
         * If role === "employee", verify userId matches session.user.id
         * If role === "manager", allow any userId
         * Upsert availability record (delete + insert, or use ON CONFLICT)
         * revalidatePath("/employees")
         * Return { success: true }
